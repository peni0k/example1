# Commands to run syntax and style checks on analysis.py

# Install required tools (if not already installed)
pip install flake8 black pytest

# Run flake8 to check PEP 8 compliance
flake8 analysis.py

# Run black to automatically format the code according to PEP 8
black analysis.py

# Run the tests to ensure functionality remains correct after formatting
pytest test_analysis.py -v

# Run flake8 on the test file as well
flake8 test_analysis.py

# Run black on the test file as well
black test_analysis.py
